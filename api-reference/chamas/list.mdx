# List Chamas

Retrieves all chama (savings group) memberships for a user, including active and past participations. This endpoint provides comprehensive information about group dynamics, contribution schedules, payout rotations, and member activity.

<Info>
Chamas are digitized traditional savings groups (stokvels) popular across Africa. Members contribute regularly and receive payouts in rotation, with blockchain transparency ensuring trust and accountability.
</Info>

## Endpoint

<CodeGroup>
```bash cURL
curl -X GET "https://api.awo-platform.com/v1/chamas?userId={userId}&status=active&includeMembers=true" \
  -H "Authorization: Bearer YOUR_ACCESS_TOKEN"
```

```javascript JavaScript SDK
import { AWOClient } from '@awo/api-sdk';

const client = new AWOClient({ 
  apiKey: 'your-api-key',
  baseURL: 'https://api.awo-platform.com/v1'
});

const chamas = await client.chamas.list({
  userId: 'user-id',
  status: 'active',
  includeMembers: true,
  includeUpcomingPayouts: true
});
```

```python Python SDK

client = AWOClient(

```
</CodeGroup>

## Query Parameters

<ParamField query="userId" type="string">
  Filter chamas by user ID (required for admin access, optional for user access)
</ParamField>

<ParamField query="status" type="string" default="all">
  Filter by chama status
  - `all` - All chamas regardless of status
  - `active` - Currently active chamas
  - `forming` - Chamas still recruiting members
  - `completed` - Finished chama cycles
  - `paused` - Temporarily inactive chamas
  - `cancelled` - Cancelled chamas
</ParamField>

<ParamField query="role" type="string" default="all">
  Filter by user's role in the chama
  - `all` - All roles
  - `admin` - User is chama administrator
  - `member` - User is regular member
  - `treasurer` - User is chama treasurer
</ParamField>

<ParamField query="contributionFrequency" type="string">
  Filter by contribution frequency
  - `weekly` - Weekly contributions
  - `monthly` - Monthly contributions
  - `quarterly` - Quarterly contributions
</ParamField>

<ParamField query="includeMembers" type="boolean" default="false">
  Include member details and activity
</ParamField>

<ParamField query="includeContributions" type="boolean" default="false">
  Include contribution history and status
</ParamField>

<ParamField query="includeUpcomingPayouts" type="boolean" default="false">
  Include upcoming payout schedule
</ParamField>

<ParamField query="includeFinancials" type="boolean" default="true">
  Include financial summary and balances
</ParamField>

<ParamField query="sortBy" type="string" default="created_date">
  Sort chamas by field
  - `created_date` - Date chama was created
  - `next_payout` - Next payout date
  - `contribution_amount` - Contribution amount
  - `member_count` - Number of members
  - `name` - Chama name (alphabetical)
</ParamField>

<ParamField query="sortOrder" type="string" default="desc">
  Sort order: `asc` (ascending) or `desc` (descending)
</ParamField>

<ParamField query="limit" type="integer" default="20">
  Maximum number of chamas to return (1-100)
</ParamField>

<ParamField query="offset" type="integer" default="0">
  Number of chamas to skip for pagination
</ParamField>

## SDK Examples

<CodeGroup>
```typescript TypeScript SDK
import { AWOClient, ChamaListOptions } from '@awo/api-sdk';

const client = new AWOClient({
  apiKey: process.env.AWO_API_KEY!,
  baseURL: 'https://api.awo-platform.com/v1'
});

class ChamaManager {
  async getUserChamas(userId: string, options?: ChamaListOptions) {
    try {
      const chamas = await client.chamas.list({
        userId,
        status: 'active',
        includeMembers: true,
        includeFinancials: true,
        includeUpcomingPayouts: true,
        ...options
      });
      
      console.log(`Found ${chamas.summary.activeChamas} active chamas`);
      console.log(`Total contributed: ${chamas.summary.totalContributed.toLocaleString()}`);
      
      return chamas;
    } catch (error) {
      console.error('Failed to get chamas:', error.message);
      throw error;
    }
  }

  async getChamasByType(userId: string, type: string) {
    const chamas = await this.getUserChamas(userId);
    return chamas.chamas.filter(chama => chama.type === type);
  }

  async getUpcomingContributions(userId: string) {
    const chamas = await this.getUserChamas(userId, { status: 'active' });
    
    const upcomingContributions = chamas.chamas
      .filter(chama => chama.userMembership.nextContributionDue)
      .map(chama => ({
        chamaId: chama.id,
        chamaName: chama.name,
        amount: chama.contributionAmount,
        dueDate: chama.userMembership.nextContributionDue,
        status: chama.userMembership.contributionStatus,
        daysUntilDue: this.calculateDaysUntilDue(chama.userMembership.nextContributionDue)
      }))
      .sort((a, b) => new Date(a.dueDate).getTime() - new Date(b.dueDate).getTime());
    
    return upcomingContributions;
  }

  async getExpectedPayouts(userId: string) {
    const chamas = await this.getUserChamas(userId, { 
      status: 'active',
      includeUpcomingPayouts: true 
    });
    
    const expectedPayouts = [];
    
    chamas.chamas.forEach(chama => {
      if (chama.type === 'rotating_savings') {
        const userPayout = chama.upcomingPayouts?.find(
          payout => payout.recipientId === userId
        );
        
        if (userPayout) {
          expectedPayouts.push({
            chamaId: chama.id,
            chamaName: chama.name,
            amount: userPayout.amount,
            scheduledDate: userPayout.scheduledDate,
            cycleNumber: userPayout.cycleNumber,
            position: chama.userMembership.payoutPosition
          });
        }
      }
    });
    
    return expectedPayouts.sort((a, b) => 
      new Date(a.scheduledDate).getTime() - new Date(b.scheduledDate).getTime()
    );
  }

  async getChamaPerformanceMetrics(userId: string) {
    const chamas = await this.getUserChamas(userId, {
      includeFinancials: true,
      includeContributions: true
    });
    
    const metrics = {
      totalChamas: chamas.summary.totalChamas,
      activeChamas: chamas.summary.activeChamas,
      totalContributed: chamas.summary.totalContributed,
      totalReceived: chamas.summary.totalReceived,
      netPosition: chamas.summary.totalReceived - chamas.summary.totalContributed,
      averageContributionCompliance: 0,
      mostActiveChama: null,
      roiAnalysis: {}
    };
    
    // Calculate average contribution compliance
    const totalPossibleContributions = chamas.chamas.reduce((sum, chama) => 
      sum + (chama.cycleCount * chama.contributionAmount), 0
    );
    
    if (totalPossibleContributions > 0) {
      metrics.averageContributionCompliance = 
        (chamas.summary.totalContributed / totalPossibleContributions) * 100;
    }
    
    // Find most active chama
    if (chamas.chamas.length > 0) {
      metrics.mostActiveChama = chamas.chamas.reduce((most, current) => 
        current.userMembership.totalContributed > most.userMembership.totalContributed 
          ? current 
          : most
      );
    }
    
    // Calculate ROI for investment clubs
    chamas.chamas
      .filter(chama => chama.type === 'investment_club')
      .forEach(chama => {
        const contributed = chama.userMembership.totalContributed;
        const interestEarned = chama.financials.interestEarned;
        const roi = contributed > 0 ? (interestEarned / contributed) * 100 : 0;
        
        metrics.roiAnalysis[chama.id] = {
          chamaName: chama.name,
          contributed,
          interestEarned,
          roi: parseFloat(roi.toFixed(2))
        };
      });
    
    return metrics;
  }

  private calculateDaysUntilDue(dueDateString: string): number {
    const dueDate = new Date(dueDateString);
    const today = new Date();
    const diffTime = dueDate.getTime() - today.getTime();
    return Math.ceil(diffTime / (1000 * 60 * 60 * 24));
  }
```

    ```typescript
    class ContributionPlanner {
      async createContributionCalendar(userId: string) {
        const chamas = await client.chamas.list({
          userId,
          status: 'active',
          includeUpcomingPayouts: true
        });
        
        const calendar = new Map();
        
        chamas.chamas.forEach(chama => {
          const nextDue = chama.userMembership.nextContributionDue;
          if (nextDue) {
            const dateKey = nextDue.split('T')[0];
            if (!calendar.has(dateKey)) {
              calendar.set(dateKey, []);
            }
            
            calendar.get(dateKey).push({
              chamaId: chama.id,
              chamaName: chama.name,
              amount: chama.contributionAmount,
              currency: chama.currency,
              frequency: chama.frequency
            });
          }
        });
        
        return Array.from(calendar.entries()).map(([date, contributions]) => ({
          date,
          contributions,
          totalAmount: contributions.reduce((sum, c) => sum + c.amount, 0)
        }));
      }
    }
    ```
